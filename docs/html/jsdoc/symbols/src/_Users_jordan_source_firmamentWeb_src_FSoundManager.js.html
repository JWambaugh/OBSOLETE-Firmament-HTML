  7 function FSoundManager(){
  8 	this.cache={};
  9 	this.players=[];
 10 	
 11 }

 13 
 14 FSoundManager.prototype = new FObservable();
 15 
 16 FSoundManager.prototype.loadSound=function(audioFile){
 17 	this.cache[audioFile]={};
 18 	this.cache[audioFile].sound = new FSound(audioFile);
 19 	//preload it 
 20 	this.cache[audioFile].player = new FSoundPlayer(this.cache[audioFile].sound);
 21 	return this.cache[audioFile].sound;
 22 }
 23 
 24 FSoundManager.prototype.getSoundPlayer=function(sound){
 25 	var soundP = new FSoundPlayer(sound);
 26 	return soundP;
 27 }
 28 
 29 
 30 FSoundManager.prototype.play=function(sound){
 31 	var soundP = this.getSoundPlayer(sound);
 32 	this.players.push(soundP);
 33 	soundP.play();
 34 	return soundP;
 35 }
