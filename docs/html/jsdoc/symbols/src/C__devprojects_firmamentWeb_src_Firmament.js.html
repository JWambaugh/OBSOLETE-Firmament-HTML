 73 		/*
 74 		 * Function: extend
 75 		 * 
 76 		 * copys one object into another
 77 		 * 
 78 		 * Parameters:
 79 		 * 		target - The object to copy into
 80 		 * 		object - The object to copy
 81 		 * 		objectcs... - More object to copy. objects will be copied from right to left.
 82 		 */
 83 		,extend:function() {
 84 			 var options, name, src, copy, copyIsArray, clone,
 85 				target = arguments[0] || {},
 86 				i = 1,
 87 				length = arguments.length,
 88 				deep = false;
 89 
 90 			// Handle a deep copy situation
 91 			if ( typeof target === "boolean" ) {
 92 				deep = target;
 93 				target = arguments[1] || {};
 94 				// skip the boolean and the target
 95 				i = 2;
 96 			}
 97 
 98 			// Handle case when target is a string or something (possible in deep copy)
 99 			if ( typeof target !== "object" && !FHelper.isFunction(target) ) {
100 				target = {};
101 			}
102 
103 		    //return the target if there's only one argument
104 			if ( length === i ) {
105 				return target
106 			}
107 
108 			for ( ; i < length; i++ ) {
109 				// Only deal with non-null/undefined values
110 				if ( (options = arguments[ i ]) != null ) {
111 					// Extend the base object
112 					for ( name in options ) {
113 						src = target[ name ];
114 						copy = options[ name ];
115 
116 						// Prevent never-ending loop
117 						if ( target === copy ) {
118 							continue;
119 						}
120 
121 						// Recurse if we're merging plain objects or arrays
122 						if ( deep && copy && ( Firmament.isPlainObject(copy) || (copyIsArray = Firmament.isArray(copy)) ) ) {
123 							if ( copyIsArray ) {
124 								copyIsArray = false;
125 								clone = src && Firmament.isArray(src) ? src : [];
126 
127 							} else {
128 								clone = src && Firmament.isPlainObject(src) ? src : {};
129 							}
130 
131 							// Never move original objects, clone them
132 							target[ name ] = Firmament.FHelper.extend( deep, clone, copy );
133 
134 						// Don't bring in undefined values
135 						} else if ( copy !== undefined ) {
136 							target[ name ] = copy;
137 						}
138 					}
139 				}
140 			}
141 
142 			// Return the modified object
143 			return target;
144 		}

155 		,isPlainObject:function( obj ) {
156 				// Must be an Object.
157 				// Because of IE, we also have to check the presence of the constructor property.
158 				// Make sure that DOM nodes and window objects don't pass through, as well
159 				if ( !obj ) {
160 					return false;
161 				}
162 				
163 				// Not own constructor property must be Object
164 				if ( obj.constructor &&
165 					!hasOwn.call(obj, "constructor") &&
166 					!hasOwn.call(obj.constructor.prototype, "isPrototypeOf") ) {
167 					return false;
168 				}
169 			
170 				// Own properties are enumerated firstly, so to speed up,
171 				// if last one is own, then all properties are own.
172 			
173 				var key;
174 				for ( key in obj ) {}
175 				
176 				return key === undefined || hasOwn.call( obj, key );
177 			}

218 Firmament.getElementOffset = function( el ) {
219     var _x = 0;
220     var _y = 0;
221     while( el && !isNaN( el.offsetLeft ) && !isNaN( el.offsetTop ) ) {
222         _x += el.offsetLeft //- el.scrollLeft;
223         _y += el.offsetTop //- el.scrollTop;
224         el = el.offsetParent;
225     }
226     return { y: _y, x: _x };
227 }
