 20 /**
 21  * A single sound asset
 22 
 23  *
 24  * @class FSound
 25  * @param {string} fileName - The name of the audio file
 26  * @extends FObservable
 27  */
 28 function FSound(fileName,duration){
 29 	
 30 	this.fileName = fileName;
 31 	this.duration=duration;
 32 	
 33 }
 34 FSound.prototype = new FObservable();
 35 
 36 
 37 /**
 38  * Gets plays the audio asset and returns the FSoundPlayer that is playing it.
 39  * @returns
 40  */
 41 FSound.prototype.play = function(){
 42 	return Firmament.getSoundManager().play(this);
