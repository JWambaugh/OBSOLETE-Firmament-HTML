 20 /**
 21  * A sound player
 22 
 23  *
 24  * @class FSoundPlayer
 25  * @param {FSound} sound - the sound object to play
 26  * @extends FObservable
 27  */
 28 function FSoundPlayer(sound){
 29 	if(sound != undefined){
 30 		this.audioObj = new Audio(sound.fileName);
 31 		this.setSound(sound);
 32 	}
 33 	else
 34 		this.audioObj = new Audio();
 35 	
 36 }
 37 
 38 
 39 
 40 
 41 FSoundPlayer.prototype = new FObservable();
 42 
 43 FSoundPlayer.prototype.setSound=function(sound){
 44 	this.soundObj=sound;
 45 	this.audioObj.src=sound.fileName;
 46 	this.audioObj.load();
 47 	
 48 };
 49 
 50 FSoundPlayer.prototype.play=function(){
 51 	
 52 	this.audioObj.play();
 53 	var duration=5;//default assume 5 second duration
 54 	if(this.audioObj.duration && ! isNaN(this.audioObj.duration));
 55 	else if(this.soundObj.duration != undefined)duration = this.soundObj.duration;
 56 	setTimeout(function(){this.emit("canCleanUp",[this])}.bind(this),(duration+1)*1000);
 57 }
