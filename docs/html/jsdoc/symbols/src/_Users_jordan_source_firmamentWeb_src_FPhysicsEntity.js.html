 24 /**
 25  * 
 26  * A an entity used in {@link FPhysicsWorld}
 27  * @class
 28  * @extends FRenderable

 31 function FPhysicsEntity(world,config){
 32 	this.world=world;
 33     this.config=config;
 34     var def = new b2BodyDef;
 35     var width=0;
 36     
 37     if(config.position){
 38         def.position.x=config.position.x;
 39         def.position.y=config.position.y;
 40     } else {
 41         def.position.x=0;
 42         def.position.y=0;
 43     }
 44     if(config.positionX!=undefined){
 45     	def.position.x=config.positionX;
 46     }
 47     if(config.positionY!=undefined){
 48     	def.position.y=config.positionY;
 49     }
 50     //def.fixedRotation=false;
 51     if(config.type=='static')def.type=b2Body.b2_staticBody;
 52     else def.type=Box2D.Dynamics.b2Body.b2_dynamicBody;
 53     if(config.angle){
 54     	def.angle=config.angle;
 55     }
 56     
 57     
 58     
 59     
 60     
 61     def.userData=this;
 62     //console.log(def);
 63     this.body=this.world.b2world.CreateBody(def);
 64     //process shape definitions
 65     for(var x=0;x<config.shapes.length;x++){
 66         var fixDef=new b2FixtureDef;
 67         var shape = config.shapes[x];
 68         
 69         if(shape.type=='box'){
 70         	shape.width= shape.width!=undefined?shape.width:1;
 71         	shape.height= shape.height!=undefined?shape.height:1;
 72         	
 73             fixDef.shape=new b2PolygonShape();
 74             fixDef.shape.SetAsBox(shape.width/2,shape.height/2);
 75             width=shape.width;
 76         } else if(shape.type=='circle'){
 77         	width=shape.radius*2;
 78             fixDef.shape = new b2CircleShape(shape.radius);
 79         }
 80         
 81         fixDef.density=shape.density!=undefined?shape.density:1;
 82         fixDef.friction=shape.friction!=undefined?shape.friction:1;
 83         fixDef.restitution=shape.restitution!=undefined?shape.restitution:0;
 84         //console.log(fixDef)
 85         this.body.CreateFixture(fixDef);
 86     }
 87    // Firmament.log(this.body);
 88     this.body.ResetMassData();
 89     //this.position=this.body.m_xf.position; //tie the entity's position to the body's position
 90     
 91     //set z value
 92     this.zPosition =0;
 93     
 94     
 95     if(config.maxLifeSeconds){
 96     	setTimeout(function(){
 97     		this.destroy();
 98     	}.bind(this),config.maxLifeSeconds*1000);
 99     }
100     
101     
102     if(config.color)this.setColor(config.color);
103     
104     
105     if(config.image){
106     	//console.log(typeof(config.image))
107     	if(typeof(config.image)=='string'){
108     		var i= document.createElement('img');
109     		i.src=config.image;
110     		config.image=i;
111     	}
112     	this.currentImage=config.image;
113     	this.setRenderer(new FSpriteRenderer);
114     	
115     	if(config.imageScale){
116         	if(config.imageScale=='auto'){
117         		if(!config.imageWidth){
118         			Firmament.log("Image width must be set for auto scale! Defaulting to 1:1");
119         			this.imageScale=100;
120         		}else {
121         			this.imageScale=config.imageWidth/width;
122         		}
123         	}else{
124         		this.imageScale=config.imageScale;
125         	}
126         }
127     	
128     	
129     }else{
130     	this.setRenderer(new FWireframeRenderer);
131     }
132     
133     
134     
135 }

139 FPhysicsEntity.prototype.getShapes=function(){
140 	var shapes = [];
141 	var s = this.body.GetFixtureList();
142 	while(s){
143 		shapes.push(s.GetShape());
144 		s = s.GetNext();
145 	}
146 	return shapes;
147 }
148 
149 FPhysicsEntity.prototype.getPosition=function(){
150 	
151 	return this.body.GetPosition();
152 }
153 
154 FPhysicsEntity.prototype.getAngle=function(){
155 	
156 	return this.body.GetAngle();
157 }
158 
159 FPhysicsEntity.prototype.getCurrentImage=function(){
160 	return this.currentImage;
161 }
162 
163 
164 
165 FPhysicsEntity.prototype.setVelocity=function(v){
166 	this.body.SetLinearVelocity(v);
167 }
168 
169 
170 FPhysicsEntity.prototype.destroy=function(){
171 	this.world.b2world.DestroyBody(this.body);
172 	this.world.destroyEntity(this)
173 }
174 
175 
176 FPhysicsEntity.prototype.deleteLater=FPhysicsEntity.prototype.destroy;
