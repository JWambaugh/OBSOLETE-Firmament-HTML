 21 /** 
 22  * @class
 23  * 
 24  * @param element  The element to attach signals to
 25  */
 26 function FInput(element){
 27 	if(element==undefined) element=document;
 28 	this.listenElement=element;
 29 	this.keysPressed={};
 30 	element.onkeyup=this._keyup.bind(this);
 31 	element.onkeydown=this._keydown.bind(this);
 32 
 33 	element.ontouchstart=this._mouseDown.bind(this);
 34 	element.ontouchend=this._mouseUp.bind(this);
 35 	element.ontouchmove=this._mouseMove.bind(this);
 36 	
 37 	element.onmousedown=this._mouseDown.bind(this);
 38 	element.onmouseup=this._mouseUp.bind(this);
 39 	element.onmousemove=this._mouseMove.bind(this);
 40 	element.selectstart=function(){return false;}
 41 	this.mouseX=0;
 42 	this.mouseY=0;
 43 	this.leftMouseDown=false;
 44 	this.rightMouseDown=false;
 45 	
 46 }
 47 
 48 FInput.prototype=new FObservable;
 49 
 50 /**
 51  * emitted when the mouse is pressed down
 52  * @name FInput#mouseDown
 53  * @event
 54  * @param {Event} e the javascript event object for the event
 55  */
 56 FInput.prototype._mouseDown=function(e){
 57 	this._updateMousePos(e);
 58 	//console.log(e)
 59 	this.leftMouseDown=true;
 60 	this.emit('mouseDown',[e]);
 61 };
 62 
 63 
 64 /**
 65  * emitted when the mouse button is released
 66  * @name FInput#mouseUp
 67  * @event
 68  * @param {Event} e the javascript event object for the event
 69  */
 70 FInput.prototype._mouseUp=function(e){
 71 	this.leftMouseDown=false;
 72 	
 73 	this._updateMousePos(e);
 74 	this.emit('mouseUp',[e]);
 75 };
 76 
 77 
 78 /**
 79  * emitted when the mouse is moved
 80  * @name FInput#mouseMove
 81  * @event
 82  * @param {Event} e the javascript event object for the event
 83  */
 84 FInput.prototype._mouseMove=function(e){
 85 	this._updateMousePos(e);
 86 	
 87 	//console.log(e);
 88 	if(this.leftMouseDown){
 89 		this.emit('mouseDrag',[e]);
 90 	}
 91 	this.emit('mouseMove', [e]);
 92 };

 95 /*
 96  * Function: getMouseScreenPos
 97  * Returns an <FVector> of the mouse's current position
 98  */
 99 FInput.prototype.getMouseScreenPos=function(e){
100 	return new FVector(this.mouseX,this.mouseY);
101 };
102 
103 /**
104  * Returns the position in the world where the mouse is currently located based on camera
105  * @param {FCamera} camera The camera
106  */
107 FInput.prototype.getMouseWorldPos=function(camera){
108 	var offset=Firmament.getElementOffset(camera.getCanvas());
109 	
110 	var x=this.mouseX-offset.x;
111 	var y=this.mouseY-offset.y;
112 	var cameraPos = camera.getTopLeftPosition();
113 	var cameraZoom=camera.getZoom();
114 	x=(this.mouseX / cameraZoom) + cameraPos.x;
115 	y=(this.mouseY / cameraZoom) + cameraPos.y;
116 	return new FVector(x,y);
117 }
118

120 	
121 FInput.prototype._updateMousePos=function(e){
122 	if(e.x!==undefined){
123 		this.mouseX=e.x;
124 		this.mouseY=e.y;
125 	}else if(e.clientX!==undefined){
126 		this.mouseX=e.clientX;
127 		this.mouseY=e.clientY;
128 	}else if(e.pageX!==undefined){
129 		this.mouseX=e.pageX;
130 		this.mouseY=e.pageY;
131 	}
132 };
133 
134 /**
135  * emitted when a key on the keyboard is released
136  * @name FInput#keyUp
137  * @event
138  * @param {int} keycode the keycode of the released key
139  * @param {Event} e the javascript event object for the event
140  */
141 FInput.prototype._keyup=function(e){
142 	var keyCode=this._getKeyCode(e);
143 	this.keysPressed[keyCode]=false;
144 	this.emit('keyUp',[keyCode,e]);
145 };
146 
147 
148 /**
149  * emitted when a key on the keyboard is pressed
150  * @name FInput#keyDown
151  * @event
152  * @param {int} keycode the keycode of the released key
153  * @param {Event} e the javascript event object for the event
154  */
155 FInput.prototype._keydown=function(e){
156 	var keyCode=this._getKeyCode(e);
157 	this.keysPressed[keyCode]=true;
158 	this.emit('keyDown',[keyCode,e]);
159 };
160 
161 
162 FInput.prototype._getKeyCode=function(e){
163 	var keynum;
164 	if(window.event) // IE
165 	{
166 		keynum = e.keyCode
167 	}
168 	else if(e.which) // Netscape/Firefox/Opera
169 	{
170 		keynum = e.which
171 	}
172 	return keynum;
173 }
174 
175 
176 FInput.prototype.isKeyPressed=function(key){
177 	if(this.keysPressed[key])return true;
178 	return false;
179 };
180 
181 
182 FInput.prototype.isMousePressed=function(button){
183 	switch(button){
184 	case 'left':
185 		return this.leftMouseDown;
186 	}
187 }
