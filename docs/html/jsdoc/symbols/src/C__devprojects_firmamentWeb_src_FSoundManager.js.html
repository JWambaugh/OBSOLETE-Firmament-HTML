 21 /**
 22  * Manages sound objects and their playback 
 23  * @class FSoundManager
 24  * 
 25  */
 26 
 27 
 28 
 29 function FSoundManager(){
 30 	this.cache={};
 31 	this.playerPool = new FResourcePool(FSoundPlayer);
 32 	
 33 }
 34 
 35 
 36 FSoundManager.prototype = new FObservable();
 37 
 38 FSoundManager.prototype.loadSound=function(audioFile,duration){
 39 	this.cache[audioFile]={};
 40 	this.cache[audioFile].sound = new FSound(audioFile,duration);
 41 	//preload it 
 42 	this.cache[audioFile].player = new FSoundPlayer(this.cache[audioFile].sound);
 43 	return this.cache[audioFile].sound;
 44 }
 45 
 46 FSoundManager.prototype.getSoundPlayer=function(sound){
 47 	var soundP = this.playerPool.get();
 48 	
 49 	soundP.disconnect();
 50 	soundP.setSound(sound);
 51 	soundP.connect("canCleanUp",function(player){this.playerPool.addBack(player)}.bind(this));
 52 	return soundP;
 53 }
 54 
 55 
 56 FSoundManager.prototype.play=function(sound){
 57 	var soundP = this.getSoundPlayer(sound);
 58 	
 59 	soundP.play();
 60 	return soundP;
 61 }
