 22 var FTriangulator={
 23 		EPSILON:0.0000000001
 24 		,area :function(contour)
 25 		{
 26 			var n = contour.length;
 27 			var A=0.0;
 28 			var p=n-1;
 29 			for(var q=0; q<n; p=q++)
 30 			{
 31 				A+= contour[p].x*contour[q].y - contour[q].x*contour[p].y;
 32 			}
 33 			return A*0.5;
 34 		}

 37 		/**
 38 		 * decides if a povar P is Inside of the triangle
 39 		 * defined by A, B, C.
 40 		 */
 41 		,insideTriangle:function(Ax,Ay,
 42 								Bx, By,
 43 								Cx, Cy,
 44 								Px, Py){
 45 			var ax, ay, bx, by, cx, cy, apx, apy, bpx, bpy, cpx, cpy;
 46 			var cCROSSap, bCROSScp, aCROSSbp;
 47 			
 48 			ax = Cx - Bx;  ay = Cy - By;
 49 			bx = Ax - Cx;  by = Ay - Cy;
 50 			cx = Bx - Ax;  cy = By - Ay;
 51 			apx= Px - Ax;  apy= Py - Ay;
 52 			bpx= Px - Bx;  bpy= Py - By;
 53 			cpx= Px - Cx;  cpy= Py - Cy;
 54 			
 55 			aCROSSbp = ax*bpy - ay*bpx;
 56 			cCROSSap = cx*apy - cy*apx;
 57 			bCROSScp = bx*cpy - by*cpx;
 58 			
 59 			return ((aCROSSbp >= 0.0) && (bCROSScp >= 0.0) && (cCROSSap >= 0.0));
 60 		}

 63 		,snip:function (contour, u, v, w, n,V)
 64 		{
 65 			  var p;
 66 			  var Ax, Ay, Bx, By, Cx, Cy, Px, Py;
 67 
 68 			  Ax = contour[V[u]].x;
 69 			  Ay = contour[V[u]].y;
 70 
 71 			  Bx = contour[V[v]].x;
 72 			  By = contour[V[v]].y;
 73 
 74 			  Cx = contour[V[w]].x;
 75 			  Cy = contour[V[w]].y;
 76 
 77 			  if ( this.EPSILON > (((Bx-Ax)*(Cy-Ay)) - ((By-Ay)*(Cx-Ax))) ) return false;
 78 
 79 			  for (p=0;p<n;p++)
 80 			  {
 81 			    if( (p == u) || (p == v) || (p == w) ) continue;
 82 			    Px = contour[V[p]].x;
 83 			    Py = contour[V[p]].y;
 84 			    if (this.insideTriangle(Ax,Ay,Bx,By,Cx,Cy,Px,Py)) return false;
 85 			  }
 86 
 87 			  return true;
 88 			}
 89 		,process:function(contour,result)
 90 		{
 91 			  /* allocate and initialize list of Vertices in polygon */
 92 
 93 			  var n = contour.length;
 94 			  if ( n < 3 ) return false;
 95 
 96 			  var V = new Array();
 97 
 98 			  /* we want a counter-clockwise polygon in V */
 99 
100 			  if ( 0.0 < this.area(contour) )
101 			    for (var v=0; v<n; v++) V[v] = v;
102 			  else
103 			    for(var v=0; v<n; v++) V[v] = (n-1)-v;
104 
105 			  var nv = n;
106 
107 			  /*  remove nv-2 Vertices, creating 1 triangle every time */
108 			  var count = 2*nv;   /* error detection */
109 
110 			  for(var m=0, v=nv-1; nv>2; )
111 			  {
112 			    /* if we loop, it is probably a non-simple polygon */
113 			    if (0 >= (count--))
114 			    {
115 			      //** Triangulate: ERROR - probable bad polygon!
116 			      return false;
117 			    }
118 
119 			    /* three consecutive vertices in current polygon, <u,v,w> */
120 			    var u = v  ; if (nv <= u) u = 0;     /* previous */
121 			    v = u+1; if (nv <= v) v = 0;     /* new v    */
122 			    var w = v+1; if (nv <= w) w = 0;     /* next     */
123 
124 			    if ( this.snip(contour,u,v,w,nv,V) )
125 			    {
126 			      var a,b,c,s,t;
127 
128 			      /* true names of the vertices */
129 			      a = V[u]; b = V[v]; c = V[w];
130 
131 			      /* output Triangle */
132 			      result.push( contour[a] );
133 			      result.push( contour[b] );
134 			      result.push( contour[c] );
135 
136 			      m++;
137 
138 			      /* remove v from remaining polygon */
139 			      for(s=v,t=v+1;t<nv;s++,t++) V[s] = V[t]; nv--;
140 
141 			      /* resest error detection counter */
142 			      count = 2*nv;
143 			    }
144 			  }
145 
146 
147 
148 			  //delete V;
149 
150 			  return true;
151 			}

153 		,getTriangles:function(shape){
154 			var res=[],triangles=[];
155 			if(!this.process(shape, res)){
156 				return false;
157 			}
158 			var triangleCount = res.length/3;
159 			for(var x=0;x<triangleCount;x++){
160 				triangles.push(
161 				[
162 				 res[x*3]
163 				 ,res[x*3+1]
164 				 ,res[x*3+2]
165 				 ]		
166 				)
167 			}
168 			return triangles;
169 		}
