 21 /** 
 22  * @class
 23  * 
 24  * @param constructor the constructor for the class to control a pool of
 25  */
 26 
 27 
 28 function FResourcePool( constructor){
 29 	//this.max=poolMax;
 30 	this.constructorFunc = constructor;
 31 	this.availablePool=[];
 32 	this.totalPool=[];
 33 
 34 }
 35 
 36 
 37 FResourcePool.prototype.get=function(){
 38 	var resource;
 39 	if(this.availablePool.length>0){
 40 		resource= this.availablePool.pop();
 41 	}else{
 42 		resource = new this.constructorFunc();
 43 		this.totalPool.push(resource);
 44 		//console.log('pool size: ' + this.totalPool.length);
 45 	}
 46 	return resource;
 47 };
 48 
 49 
 50 FResourcePool.prototype.addBack=function(object){
 51 	//adds a resource back to the pool
 52 	this.availablePool.push(object)
 53 };
