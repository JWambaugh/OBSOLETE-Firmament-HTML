 18 /**
 19  * @class FCamera
 20  */
 21 function FCamera(canvas){
 22 	this.canvas=canvas;
 23 	this.width=canvas.width;
 24 	this.height=canvas.height;
 25 
 26 
 27 	this.game= null;
 28 	this.zoom=100;
 29 	this.topLeftPosition=new FVector();
 30 	this.setPosition(new FVector(0,0));
 31 }
 32 
 33 FCamera.prototype=new FWorldPositional;
 34 
 35 
 36 FCamera.prototype.render=function(worlds){
 37 	var cxt = this.getCanvas().getContext('2d');
 38 	cxt.clearRect(0, 0, this.width, this.height);
 39 
 40 	this.emit('beginRender',[cxt]);
 41 	for(var x=0;x<worlds.length;x++){
 42 		var world = worlds[x];
 43 		//just grab entities that are inside the camera for rendering
 44 		var entities=world.getEntitiesInBox(this.position.x-this.width/2/this.zoom,this.position.y-this.height/2/this.zoom,this.position.x+this.width/2/this.zoom,this.position.y+this.height/2/this.zoom);
 45 		//Firmament.log(entities);
 46 		for(var y=0;y<entities.length;y++){
 47 			var ent = entities[y];
 48 			ent.getRenderer().render(cxt,ent,this);
 49 		}
 50 	}
 51 	
 52 	this.emit('endRender',[cxt]);
 53 };
 54 
 55 
 56 FCamera.prototype.getCanvas = function(){
 57 	return this.canvas;
 58 }
 59 
 60 FCamera.prototype.setGame=function(g){
 61 	this.game = g;
 62 };
 63 
 64 
 65 /**
 66  * Sets the width of the camera's display
 67  * @param {FVector} width
 68  */
 69 
 70 FCamera.prototype.setWidth=function(w){
 71     this.width=w;
 72     this.calculateTopLeftPosition();
 73 };
 74 
 75 FCamera.prototype.getWidth=function(){
 76 	return this.width;
 77 }
 78 
 79 /**
 80  * Sets the width of the camera's display
 81  * @param {FVector} width
 82  */
 83 
 84 FCamera.prototype.setHeight=function(h){
 85     this.height=h;
 86     this.calculateTopLeftPosition();
 87 };
 88 
 89 FCamera.prototype.getHeight=function(){
 90 	return this.height;
 91 }
 92 
 93 
 94 FCamera.prototype.getZoom=function(){
 95 	return this.zoom;
 96 }
 97 
 98 FCamera.prototype.setZoom=function(z){
 99 	this.zoom=z;
100 	this.calculateTopLeftPosition();
101 }
102 
103 
104 
105 FCamera.prototype.setPosition=function(pos){
106 	this.position=pos;
107 	this.calculateTopLeftPosition();
108 }
109 
110 FCamera.prototype.getTopLeftPosition=function(){
111 	return this.topLeftPosition;
112 }
113 
114 FCamera.prototype.calculateTopLeftPosition=function(){
115 	this.topLeftPosition.x=this.position.x-(this.width/this.zoom)/2;
116 	this.topLeftPosition.y=this.position.y-(this.height/this.zoom)/2;
117 }
