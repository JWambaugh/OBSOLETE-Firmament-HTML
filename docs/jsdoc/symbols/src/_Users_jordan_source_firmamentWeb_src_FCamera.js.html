  3 /**
  4  * Class: FCamera
  5  */
  6 function FCamera(canvas){
  7 	this.canvas=canvas;
  8 	this.width=canvas.width;
  9 	this.height=canvas.height;
 10 
 11 
 12 	this.game= null;
 13 	this.zoom=100;
 14 	this.topLeftPosition=new FVector();
 15 	this.setPosition(new FVector(0,0));
 16 }
 17 
 18 FCamera.prototype=new FWorldPositional;
 19 
 20 
 21 FCamera.prototype.render=function(worlds){
 22 	var cxt = this.getCanvas().getContext('2d');
 23 	cxt.clearRect(0, 0, this.width, this.height);
 24 
 25 	this.emit('beginRender',[cxt]);
 26 	for(var x=0;x<worlds.length;x++){
 27 		var world = worlds[x];
 28 		//just grab entities that are inside the camera for rendering
 29 		var entities=world.getEntitiesInBox(this.position.x-this.width/2/this.zoom,this.position.y-this.height/2/this.zoom,this.position.x+this.width/2/this.zoom,this.position.y+this.height/2/this.zoom);
 30 		//Firmament.log(entities);
 31 		for(var y=0;y<entities.length;y++){
 32 			var ent = entities[y];
 33 			ent.getRenderer().render(cxt,ent,this);
 34 		}
 35 	}
 36 	
 37 	this.emit('endRender',[cxt]);
 38 };

 99 FCamera.prototype.calculateTopLeftPosition=function(){
100 	this.topLeftPosition.x=this.position.x-(this.width/this.zoom)/2;
101 	this.topLeftPosition.y=this.position.y-(this.height/this.zoom)/2;
