  2 /**
  3  * Class: FEntityRepo
  4  * Singleton registry for storing element types
  5  */
  6 
  7 
  8 var FEntityRepo={
  9     
 10     elements:{}
 11     
 12     /**
 13      * registers a new entity type in the registry.
 14      * @param {string} name the name of the new entity type
 15      * @param {Object} type an {@link FEntity} config object for the new type
 16      */
 17     ,addEntityType:function(name,type){
 18     	var completedShapes=[];
 19     	for(var x=0;x<type.shapes.length;x++){
 20     		var shape =type.shapes[x];
 21     		if(shape.type=='polygon'){
 22     			var v=FTriangulator.getTriangles(shape.vectors);
 23     			if(v!==false) {
 24     				for(var y=0;y<v.length;y++){
 25     					var ob = this.clone(shape);
 26     					ob.vectors = v[y];
 27     					ob.type='triangle';
 28     					completedShapes.push(ob);
 29     				}
 30     			}
 31     		}else{
 32     			completedShapes.push(shape)
 33     		}
 34     	}
 35     	type.shapes=completedShapes;
 36         this.elements[name]=type;
 37     }
 38     
 39     
 40     /**
 41      * Returns an {@link FEntity} config object from the registry of the specified type
 42      * @param {String} name the name of the type of  {@link FEntity} config object to return
 43      * @return {Object} The config object to create a {@link FEntity} object.
 44      */
 45     ,getEntityType:function(name){
 46         var ob=this.clone(this.elements[name]);
 47         ob._entityTypeName=name;
 48         return ob;
 49     }
 50     /**
 51      * Creates a new {@link FEntity} of type name and returns it.
 52      * This is a helper function that uses getEntityType in conjuction with {@link FGame#createEntity}.
 53      * @param {String} name the name of the {@link FEntity} type to create
 54      */
 55     ,createEntity:function(name){
 56         
 57         return FGame.createEntity(this.getEntityType(name));
 58     }
 59     
 60     ,clone:function (o) {
 61     	function c(o) {
 62     		for (var i in o) {
 63     		this[i] = o[i];
 64     		}
 65     		}
 66 
 67     		return new c(o);
 68     		}
 69     
 70 };
