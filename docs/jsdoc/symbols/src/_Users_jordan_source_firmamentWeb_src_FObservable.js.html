 14 FObservable.prototype.connect=function(eventName,func,scope){
 15 	if(this._connections==undefined)this._connections={};
 16 	if(this._connections[eventName] == undefined){
 17 		this._connections[eventName]=[];
 18 	}
 19 	if(scope==undefined)scope=this;
 20 	this._connections[eventName].push({
 21 			func:func
 22 			,scope:scope
 23 		});
 24 }
 25 
 26 FObservable.prototype.disconnect=function(eventName,func){
 27 	if(this._connections==undefined)this._connections={};
 28 	//only remove specified function
 29 	if(func != undefined){
 30 		var connections=this._connections[eventName];
 31 		if(connections != undefined){
 32 			for(var x=0;x<connections.length;x++){
 33 				if(connections[x]==func){
 34 					connections.splice(x,1);
 35 				}
 36 			}
 37 		}
 38 	} else{
 39 		//remove all functions connected to event
 40 		this._connections[eventName]=[];
 41 	}
 42 }

 44 
 45 FObservable.prototype.emit=function(eventName,params){
 46 	if(this._connections==undefined)this._connections={};
 47 	var connections=this._connections[eventName];
 48 	if(connections != undefined){
 49 		for(var x=0;x<connections.length;x++){
 50 			connections[x].func.apply(connections[x].scope,params);
 51 		}
 52 	}
 53 	
 54 }
