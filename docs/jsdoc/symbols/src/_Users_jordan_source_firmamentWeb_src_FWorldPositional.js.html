 22 /**
 23  * A positional object has a position in the game world.
 24  * @class
 25  * @extends FObservable
 26  */
 27 function FWorldPositional(){
 28 	this.position = new FVector(0,0);
 29     this.positionBase='w'; //'w' = world based, 'c' = camera based
 30     this.angle=0;
 31 	
 32 }
