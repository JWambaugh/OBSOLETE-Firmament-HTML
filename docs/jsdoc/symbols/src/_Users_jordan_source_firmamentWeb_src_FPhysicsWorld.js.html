  6 function FPhysicsWorld(gravity){
  7 	this.collisions=[]
  8 	this.b2world = new b2World(
  9             new b2Vec2(gravity.x, gravity.y)    //gravity
 10          ,  true                 //allow sleep
 11       );
 12 	this.b2world.SetContactListener({
 13 		BeginContact:function(contact){
 14 			this.collisions.push(contact);
 15 		}.bind(this)
 16 		,EndContact:function(){}
 17 		,PreSolve:function(){}
 18 		,PostSolve:function(){}
 19 	})
 20 }

 28 FPhysicsWorld.prototype.step=function(fpsGoal){
 29 	this.collisions=[];
 30 	this.b2world.Step(
 31             1 / fpsGoal   //frame-rate
 32          ,  10       //velocity iterations
 33          ,  10       //position iterations
 34       );
 35 	for(var x=0;x<this.collisions.length;x++){
 36 		var c=this.collisions[x];
 37 		var ent1=c.m_fixtureA.m_body.m_userData;
 38 		var ent2=c.m_fixtureB.m_body.m_userData;
 39 		
 40 		ent1.emit("collide",[ent2,c]);
 41 		ent2.emit("collide",[ent1,c]);
 42 		//console.log("collission!");
 43 		//console.log(c);
 44 		
 45 	}
 46 	//this.world.DrawDebugData();
 47 	this.b2world.ClearForces();
 48 }

 54 FPhysicsWorld.prototype.createEntity=function(config){
 55 	//Firmament.log(this);
 56     var ent= new FPhysicsEntity(this,config);
 57     this.addEntity(ent);
 58     if(config.init){
 59     	//Firmament.log(ent._connections);
 60     	config.init.apply(ent,[]);
 61     }
 62     return ent;
 63 };

 67 FPhysicsWorld.prototype.getEntitiesInBox=function(topLeftX,topLeftY,bottomRightX,bottomRightY){
 68 	var selectEntities=[];
 69 	var query = new Box2D.Collision.b2AABB;
 70 	
 71 	query.upperBound.Set(bottomRightX,bottomRightY);
 72     query.lowerBound.Set(topLeftX,topLeftY);
 73     //Firmament.log(query,true);
 74     //Firmament.log(query);
 75     this.b2world.QueryAABB(function(fixture){
 76     	//Firmament.log("here");
 77     	selectEntities.push(fixture.GetBody().GetUserData());
 78     	return true;
 79     },query);
 80    // Firmament.log(selectEntities.length);
 81     return selectEntities;
 82 }
