 30 FSpriteRenderer.prototype = new FRenderer;
 31 
 32 var renderCount=0;
 33 FSpriteRenderer.prototype.render = function(cxt,item,camera){
 34 	var cameraPos=camera.getTopLeftPosition();
 35 	var pos = item.getPosition();
 36 	
 37 	//make sure item is within rendering range
 38 	/*if(pos.x < cameraPos.x-2 || pos.x >cameraPos.x+camera.getWidth()+2)return;
 39 	if(pos.y < cameraPos.y-2 || pos.y >cameraPos.y+camera.getHeight()+2)return;*/
 40 	
 41 	
 42 	
 43 	var shapes=item.getShapes();
 44 	
 45 	var bodyAngle=item.getAngle();
 46 	var image=item.getCurrentImage();
 47 	var ratio=camera.getZoom()/item.getImageScale();
 48 	
 49 	var a=1,b=0,c=0,d=1,tx=0,ty=0;
 50 	cxt.save();
 51 	
 52 	
 53 	
 54 	
 55 	
 56 	
 57 	if(ratio!=1)
 58 		cxt.scale(ratio,ratio);
 59 	
 60 	
 61 	var scaleRatio = camera.getZoom()/ratio;
 62 	if(bodyAngle!=0){
 63 		
 64 		
 65 		//rotate our matrix
 66 		var _a=a;
 67 		var _b=b;
 68 		var _c=c;
 69 		var _d=d;
 70 		var _tx=tx;
 71 		var _ty=ty;
 72 		var sin=Math.sin(bodyAngle);
 73 		var cos=Math.cos(bodyAngle);
 74 		a=_a*cos-_b*sin;
 75 		b=_a*sin + _b*cos;
 76 		c = _c*cos - _d*sin;
 77 		d = _c*sin +_d*cos;
 78 		tx=_tx*cos-_ty*sin;
 79 		ty =_tx*sin+_ty*cos;
 80 		
 81 		
 82 		//apply rotation
 83 		cxt.transform(a,b,c,d,tx,ty);
 84 		
 85 		
 86 		
 87 		//get inverse of matrix
 88 		
 89 		var adbc = (a*d-b*c);
 90 		var ia=d/adbc;
 91 		var ib=-b/adbc;
 92 		var ic=-c/adbc;
 93 		var id=a/adbc;
 94 		var itx=(c*ty-d*tx)/adbc;
 95 		var ity=(a*ty-b*tx)/adbc;
 96 		
 97 		
 98 		
 99 		//get x and y in relation to the inverted matrix
100 		var x=(pos.x-cameraPos.x)*scaleRatio;
101 		var y=(pos.y-cameraPos.y)*scaleRatio;
102 		var nx=ia*x+ic*y+itx;
103 		var ny=id*y+ib*x+ity;
104 		
105 	} else {
106 		var nx=(pos.x-cameraPos.x)*scaleRatio;
107 		var ny=(pos.y-cameraPos.y)*scaleRatio;
108 	}
109 	
110 	cxt.drawImage(image,nx-image.width/2,ny-image.height/2);
111 	cxt.restore();
112 	//console.log(shapes);
113 	renderCount++;
114 }
