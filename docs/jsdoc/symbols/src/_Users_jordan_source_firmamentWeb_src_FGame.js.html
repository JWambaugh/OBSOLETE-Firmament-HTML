 38 FGame.prototype=new FObservable;

 40 
 41 FGame.prototype.startSimulation=function(){
 42 	 this.stepInterval=window.setInterval(this.step.bind(this), 1000 / this.fpsGoal);
 43 };
 44 
 45 
 46 FGame.prototype.stopSimulation=function(){
 47 	 window.clearInterval(this.stepInterval);
 48 };

 51 FGame.prototype.addCamera=function(camera){

 68 FGame.prototype.step=function() {
 69 	if(this.instep)return;
 70 	this.instep=true;
 71 	
 72 	if(this.fps>0&&this.fps<10){
 73 		Firmament.log(this.worlds);
 74 	}
 75 	this.emit("beginStep");
 76 	for(var x=0;x<this.worlds.length;x++){
 77 		this.worlds[x].step(this.fpsGoal);
 78 	}
 79 	this.emit("endStep");
 80 	//window.console.log(this.world);
 81 	//call render on all cameras
 82 	
 83 	this.emit("endRender");
 84 	for(var x=0; x<this.cameras.length;x++){
 85 		this.cameras[x].render(this.worlds);
 86 	}
 87 	this.emit("endRender");
 88 	this.frames++;
 89 	this.instep=false;
 90   };
 91   
 92   FGame.prototype.frameCount=function(){
 93 	  this.fps=this.frames;
 94 	  this.emit("fpsUpdate",[this.fps]);
 95 	  this.frames=0;
 96   }
