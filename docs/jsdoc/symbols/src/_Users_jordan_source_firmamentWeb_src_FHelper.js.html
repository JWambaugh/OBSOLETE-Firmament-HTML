 24  * Shoots an entity of type [bulletDef] from the entity [origin] towards the current mouse position
 25  * @param {FEntity} origin The entity to shoot the bullet from
 26  * @param {FEntityConfig} bulletDef A config object for the entity to shoot towards the mouse 
 27  */
 28 FHelper.shootBulletFromEntityToMouse=function(input,camera,world,entity,bulletDef){
 29 	
 30     //get angle from entity to mouse
 31 	var mouseWorldPos = input.getMouseWorldPos(camera);
 32 	var entityPos = entity.getPosition();
 33     var xdiff=mouseWorldPos.x-entityPos.x;
 34     var ydiff=mouseWorldPos.y-entityPos.y;
 35     var angle=Math.atan2(ydiff,xdiff);
 36     bulletDef.positionX=entityPos.x+Math.cos(angle)*1.1;
 37     bulletDef.positionY=entityPos.y+Math.sin(angle)*1.1   ;
 38     /*console.log(angle)
 39     console.log(mouseWorldPos)
 40     console.log(JSON.stringify(entityPos))*/
 41     var bullet=world.createEntity(bulletDef);
 42     
 43     bullet.setVelocity({x:Math.cos(angle)*10,y:Math.sin(angle)*10});
 44     
 45 }

117 /**
118  * Returns an appropriate value for an entity's imageScale property based on the values passed.
119  * @param {object} params a config object with the following properties:<ul>
120  *      <li><b>imageWidth</b> - the width of the image (required)</li>
121  *      <li><b>entityWidth</b> - the width of the image (this or entity radius are required)</li>
122  *      <li><b>entityRadius</b> - the radius of the entitity</li></ul>
123  * @return {float} The proper scale value to make the image the same size as the entity
124  */
125 FHelper.getImageScale=function(params){
126     if(typeof(params)!='object'){
127         throw("Param is not an object")
128     }
129     if(params.imageWidth && params.entityWidth){
130         return params.imageWidth/params.entityWidth;
131     }else if(params.imageWidth && params.entityRadius){
132         return params.imageWidth/(params.entityRadius*2);
133     }else{
134         throw("required parameters imageWidth and either entityRadius or entityWidth missing.");
135     }
136 }

140 /**
141  *
142  *  converts style object into css. 
143  *  allows for easy creation of styles in json syntax.
144  *  @note Javascript does not allow dashes in proprty names, so properties like 'font-weight' must be written in camel case: 'fontWeight'.
145  *      To allow you to still use these properties, this function automatically converts 'fontWeight' to 'font-weight'.
146  *  @example fh.css({
147         color:'red'
148         ,background:'transparent'
149         ,fontWeight:'bold'
150         ,fontSize:'20px'
151     })    
152  *  
153 **/
154 FHelper.css=function(object){
155     
156     function fixKey(key){
157        var c=/[A-Z]/.exec(key)
158        while(c&&typeof(c[0])!='undefined'){
159             key=key.split(c[0]).join('-'+c[0].toLowerCase())
160             c=/[A-Z]/.exec(key)
161         }
162         return key;
163     }
164     
165     if(typeof(object)!='object'){
166         return ''
167     }
168     var buffer='';
169     for(key in object){
170         buffer+=fixKey(key)+':'
171         buffer+=object[key]+";"
172     }
173     return buffer;
174 }

179  * Handy function for debugging objects. Returns a string representation of the object provided
180  * @param {mixed}
181  * @return {string}
182  */
183 
184 FHelper.debug=function(object, indent){
185     if(!indent)indent=0;
186     var buffer="";
187     for(i in object){
188         buffer+="\n";
189         for(var x=0;x<indent;x++){
190             buffer+=" "
191         }
192         if(typeof (object[i]) == 'object'){
193             buffer=buffer+i+"->"+FHelper.debug(object[i],indent+5);
194         } else 
195         buffer=buffer+i+"->"+object[i];
196     }
197     return buffer;
198 }

200 FHelper.extend=function() {
201 	 var options, name, src, copy, copyIsArray, clone,
202 		target = arguments[0] || {},
203 		i = 1,
204 		length = arguments.length,
205 		deep = false;
206 
207 	// Handle a deep copy situation
208 	if ( typeof target === "boolean" ) {
209 		deep = target;
210 		target = arguments[1] || {};
211 		// skip the boolean and the target
212 		i = 2;
213 	}
214 
215 	// Handle case when target is a string or something (possible in deep copy)
216 	if ( typeof target !== "object" && !FHelper.isFunction(target) ) {
217 		target = {};
218 	}
219 
220     //return the target if there's only one argument
221 	if ( length === i ) {
222 		return target
223 	}
224 
225 	for ( ; i < length; i++ ) {
226 		// Only deal with non-null/undefined values
227 		if ( (options = arguments[ i ]) != null ) {
228 			// Extend the base object
229 			for ( name in options ) {
230 				src = target[ name ];
231 				copy = options[ name ];
232 
233 				// Prevent never-ending loop
234 				if ( target === copy ) {
235 					continue;
236 				}
237 
238 				// Recurse if we're merging plain objects or arrays
239 				if ( deep && copy && ( FHelper.isPlainObject(copy) || (copyIsArray = FHelper.isArray(copy)) ) ) {
240 					if ( copyIsArray ) {
241 						copyIsArray = false;
242 						clone = src && FHelper.isArray(src) ? src : [];
243 
244 					} else {
245 						clone = src && FHelper.isPlainObject(src) ? src : {};
246 					}
247 
248 					// Never move original objects, clone them
249 					target[ name ] = FHelper.extend( deep, clone, copy );
250 
251 				// Don't bring in undefined values
252 				} else if ( copy !== undefined ) {
253 					target[ name ] = copy;
254 				}
255 			}
256 		}
257 	}
258 
259 	// Return the modified object
260 	return target;
261 };

270 FHelper.isPlainObject=function( obj ) {
271 		// Must be an Object.
272 		// Because of IE, we also have to check the presence of the constructor property.
273 		// Make sure that DOM nodes and window objects don't pass through, as well
274 		if ( !obj ) {
275 			return false;
276 		}
277 		
278 		// Not own constructor property must be Object
279 		if ( obj.constructor &&
280 			!hasOwn.call(obj, "constructor") &&
281 			!hasOwn.call(obj.constructor.prototype, "isPrototypeOf") ) {
282 			return false;
283 		}
284 	
285 		// Own properties are enumerated firstly, so to speed up,
286 		// if last one is own, then all properties are own.
287 	
288 		var key;
289 		for ( key in obj ) {}
290 		
291 		return key === undefined || hasOwn.call( obj, key );
292 	}

343 /**
344  * Builds an appropriate shapes array for an entity that contains a set of triangle shapes to make up the complex polygon shape.
345  * Use this for building entities with shapes that would otherwise not be supported, such as concave polygons.
346  * Example useage:
347  * 
348  *   fgame.createEntity({
349  *                positionX:2
350  *               ,positionY:2
351  *               ,type:'static'
352  *               ,shapes:FHelper.polyShape([
353  *                   {x:0,y:0}
354  *                   ,{x:0,y:1}
355  *                   ,{x:1,y:1}
356  *                   ,{x:2,y:2}
357  *                   ,{x:2,y:1}
358  *                   ,{x:1,y:0}
359  *               ],{})
360  *               ,init:function(){
361  *               }
362  *           })
363  */
364 FHelper.polyShape=function(points,config){
365     
366     shapes=FGame.triangulateShape(points);
367     var s=[];
368     for(x=0;x<shapes.length;x++){
369         s[x]=FHelper.extend({},config,{
370             type:"polygon"
371             ,vectors:shapes[x]
372         })
373     }
374     return s;
375 }

387     for(var x=0;x<entity.shapes.length;x++){
388         var shape=entity.shapes[x];

408 /**
409  * only works on polygon shapes
410  *
411  */
412 FHelper.explodeEntity=function(entity,entConfig,subdivideCount){
413     for(var x=0;x<entity.shapes.length;x++){
414         var shape=entity.shapes[x];
415         //deactivate the parent entity
416         //entity.setActive(false);
417         
418         
419         if(shape.type=='polygon'){
420             var vectors=shape.vectors;
421             for(var i=0;i<subdivideCount;i++){
422                 vectors=FGame.subdivideShape(vectors);
423             }
424             var shapes=FGame.triangulateShape(vectors);
425             for(var j=0;j<shapes.length;j++){
426                 
427                 FGame.createEntity(FHelper.extend({},entConfig,{
428                     positionX:entity.getPositionX()
429                     ,positionY:entity.getPositionY()
430                     ,angle:entity.getAngle()
431                     ,shapes:FHelper.polyShape(shapes[j],{})
432                 }))
433             }
434         }
435     }
436     entity.deleteLater();
437 }

566 /**
567  * Finds entities that have a property with the specified value.
568  * @param {String} property the name of the property to search
569  * @param {String} value the value to match against
570  * @return [Array] array of matching entities
571  */
572 FHelper.findEntities=function(property,value){
573     var entities=fg.getAllEntities();
574     var matches=[];
575     for(var x=0;x<entities.length;x++){
576         if(entities[x][property]==value){
577             matches.push(entities[x]);
578         }
579     }
580     return matches;
581 }
