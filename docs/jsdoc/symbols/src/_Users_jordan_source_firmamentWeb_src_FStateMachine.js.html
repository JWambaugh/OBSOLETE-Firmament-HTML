  2 /**
  3  * @class FStateMachine
  4  * Base class for creating a state machine
  5  */
  6 
  7 
  8 function FStateMachine(states){
  9     this.states=states;
 10     this.currentState=null;
 11     this.currentStateId=null;
 12     
 13     this.setState=function(stateId){
 14         //do nothing if we're already in that state
 15         if(stateId==this.currentStateId) return;
 16         if(this.currentState!=null && typeof(this.stopState)=='function'){
 17             this.currentState.stopState.call(this);
 18         }
 19         if(this.states[stateId]){
 20             this.currentState=this.states[stateId]
 21             this.currentStateId=stateId;
 22             if(typeof(this.currentState.initState)=='function'){
 23                 this.currentState.initState.call(this);
 24             }
 25         } else {
 26             throw "State of type "+stateId+' Is not a valid state object'
 27         }
 28     }
 29     
 30     /**
 31      *
 32      * Returns the currentState
 33      */
 34     this.getState=function(){
 35         return this.currentState;
 36     }
 37     
 38     /**
 39      * Calls the function of name func in the current state, in the scope of the state machine.
 40      * @param {function} func the name of the function to call
 41      * @param {Array} arguments the arguments to pass to the function
 42      * @note For the sake of speed, this function does not do any error checking.
 43      *
 44      */
 45     this.callState=function(func,args){
 46         this.currentState[func].apply(this,args)
 47     }
 48     
 49 }
