  2 /*
  3  * Class: FInput
  4  * Provides input signals for a specified html element
  5  * 
  6  * Extends: <FObservable>
  7  */
  8 
  9 /*
 10  * Constructor: FInput
 11  * Parameters: 
 12  * 		element - The element to attach signals to
 13  */
 14 function FInput(element){
 15 	if(element==undefined) element=document;
 16 	this.listenElement=element;
 17 	this.keysPressed={};
 18 	element.onkeyup=this._keyup.bind(this);
 19 	element.onkeydown=this._keydown.bind(this);
 20 
 21 	element.ontouchstart=this._mouseDown.bind(this);
 22 	element.ontouchend=this._mouseUp.bind(this);
 23 	element.ontouchmove=this._mouseMove.bind(this);
 24 	
 25 	element.onmousedown=this._mouseDown.bind(this);
 26 	element.onmouseup=this._mouseUp.bind(this);
 27 	element.onmousemove=this._mouseMove.bind(this);
 28 	element.selectstart=function(){return false;}
 29 	this.mouseX=0;
 30 	this.mouseY=0;
 31 	this.leftMouseDown=false;
 32 	this.rightMouseDown=false;
 33 	
 34 }
 35 
 36 FInput.prototype=new FObservable;

 39 /*
 40  * Signal: mouseDown
 41  * emitted when the mouse is pressed down
 42  * 
 43  * Parameters:
 44  * 		e - the javascript event object for the event
 45  */
 46 
 47 FInput.prototype._mouseDown=function(e){
 48 	this._updateMousePos(e);
 49 	//console.log(e)
 50 	this.leftMouseDown=true;
 51 	this.emit('mouseDown',[e]);
 52 };
 53 
 54 
 55 /*
 56  * Signal: mouseUp
 57  * emitted when the mouse button is released
 58  * 
 59  * Parameters:
 60  * 		e - the javascript event object
 61  */
 62 
 63 
 64 FInput.prototype._mouseUp=function(e){
 65 	this.leftMouseDown=false;
 66 	
 67 	this._updateMousePos(e);
 68 	this.emit('mouseUp',[e]);
 69 };

 72 /*
 73  * Signal: mouseMove
 74  * emitted when the mouse is moved
 75  * 
 76  * Parameters:
 77  * 		e - the javascript event object
 78  */
 79 FInput.prototype._mouseMove=function(e){
 80 	this._updateMousePos(e);
 81 	
 82 	//console.log(e);
 83 	if(this.leftMouseDown){
 84 		this.emit('mouseDrag',[e]);
 85 	}
 86 	this.emit('mouseMove', [e]);
 87 };

 90 /*
 91  * Function: getMouseScreenPos
 92  * Returns an <FVector> of the mouse's current position
 93  */
 94 FInput.prototype.getMouseScreenPos=function(e){
 95 	return new FVector(this.mouseX,this.mouseY);
 96 };

 98 /*
 99  * Function: getMouseWorldPos
100  * Returns the position in the world where the mouse is currently located based on camera
101  * Parameters: camera <FCamera> - The camera
102  */
103 FInput.prototype.getMouseWorldPos=function(camera){
104 	var offset=Firmament.getElementOffset(camera.getCanvas());
105 	
106 	var x=this.mouseX-offset.x;
107 	var y=this.mouseY-offset.y;
108 	var cameraPos = camera.getTopLeftPosition();
109 	var cameraZoom=camera.getZoom();
110 	x=(this.mouseX / cameraZoom) + cameraPos.x;
111 	y=(this.mouseY / cameraZoom) + cameraPos.y;
112 	return new FVector(x,y);
113 }

117 FInput.prototype._updateMousePos=function(e){
118 	if(e.x!==undefined){
119 		this.mouseX=e.x;
120 		this.mouseY=e.y;
121 	}else if(e.clientX!==undefined){
122 		this.mouseX=e.clientX;
123 		this.mouseY=e.clientY;
124 	}else if(e.pageX!==undefined){
125 		this.mouseX=e.pageX;
126 		this.mouseY=e.pageY;
127 	}
128 };

130 FInput.prototype._keyup=function(e){
131 	var keyCode=this._getKeyCode(e);
132 	this.keysPressed[keyCode]=false;
133 	this.emit('keyUp',[keyCode,e]);
134 };
135 
136 
137 
138 FInput.prototype._keydown=function(e){
139 	var keyCode=this._getKeyCode(e);
140 	this.keysPressed[keyCode]=true;
141 	this.emit('keyDown',[keyCode,e]);
142 };
143 
144 
145 FInput.prototype._getKeyCode=function(e){
146 	var keynum;
147 	if(window.event) // IE
148 	{
149 		keynum = e.keyCode
150 	}
151 	else if(e.which) // Netscape/Firefox/Opera
152 	{
153 		keynum = e.which
154 	}
155 	return keynum;
156 }

159 FInput.prototype.isKeyPressed=function(key){
160 	if(this.keysPressed[key])return true;
161 	return false;
162 };

165 FInput.prototype.isMousePressed=function(button){
166 	switch(button){
167 	case 'left':
168 		return this.leftMouseDown;
169 	}
170 }
